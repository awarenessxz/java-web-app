# Usage

To use this component, do the following:

1. Add the library as dependency
    - `yarn add react-component-library`
2. Import the component
    ```javascript
    import React, { useState } from 'react';
    import { TransferList, SelectOption } from 'react-component-library';

    ...
    const options = [
        { key: '1', value: 'text1' },
        { key: '2', value: 'text2' },
    ];
    const [selected, setSelected] = useState<SelectOption[]>([]);

    const handleTransferChange = (srcList: SelectOption[], destList: SelectOption[]): void => {
        console.log(srcList, destList);
        setSelected(destList);
    };

    return (
        <TransferList
            options={options}
            selectedOptions={selected}
            onTransfer={handleTransferChange}
        />
    );
    ...
    ```
